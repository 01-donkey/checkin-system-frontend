      // 🌟 獲取所有打卡紀錄
      async function fetchRecords() {
        tableBody.innerHTML = '<tr><td colspan="7" class="p-8 text-center text-gray-400 font-medium animate-pulse">🔄 載入資料中...</td></tr>';
        try {
          const res = await fetch(`${API_URL}/api/records`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ password: currentPassword })
          });
          const data = await res.json();
          
          if (data.success && data.records.length > 0) {
            const groupedData = {};
            const locationsSet = new Set(); // 🌟 用來收集到底有幾個不重複的場地

            data.records.forEach(r => {
              const dateObj = new Date(r.timestamp);
              const dateStr = dateObj.toLocaleDateString('zh-TW', { timeZone: 'Asia/Taipei', month: '2-digit', day: '2-digit' });
              const fullDateStr = dateObj.toLocaleDateString('zh-TW', { timeZone: 'Asia/Taipei' }); 
              const timeStr = dateObj.toLocaleTimeString('zh-TW', { timeZone: 'Asia/Taipei', hour: '2-digit', minute: '2-digit' });
              
              const key = `${r.worker_name}_${dateStr}`;

              locationsSet.add(r.location_name); // 🌟 收集場地名稱

              if (!groupedData[key]) {
                groupedData[key] = {
                  workerId: r.worker_id, // 🌟 【新增】：偷偷把員工 ID 記下來
                  date: dateStr,
                  fullDate: fullDateStr,
                  group: r.worker_group || '未分類',
                  subGroup: r.sub_group || '無',
                  name: r.worker_name,
                  location: r.location_name,
                  inTime: '<span class="text-gray-300">--:--</span>',
                  outTime: '<span class="text-gray-300">--:--</span>',
                  hasIn: false,
                  hasOut: false,
                  bento: '', // 🌟 把原本的 workLog 改成便當
                  specialStatus: ''

                  //記得加,如果後面還有欄位要加的話
                };
              }

              if (r.action === 'IN' && !groupedData[key].hasIn) {
                groupedData[key].inTime = `<span class="font-bold text-emerald-700">${timeStr}</span>`;
                groupedData[key].hasIn = true;
                // 🌟 【關鍵修改】：抓取上班時選好的便當
                if (r.work_log) groupedData[key].bento = r.work_log; 
              } else if (r.action === 'OUT') {
                if (r.special_status) groupedData[key].specialStatus = r.special_status;
                groupedData[key].outTime = `<span class="font-bold text-rose-700">${timeStr}</span>`;
                groupedData[key].hasOut = true;
                // 🌟 下班就不抓 work_log 了
              }
            });

            // 🌟 自動產生場地的下拉選項 (保留目前的選擇)
            const currentSelected = locationFilter.value;
            locationFilter.innerHTML = '<option value="ALL">顯示全部場地</option>';
            locationsSet.forEach(loc => {
              const opt = document.createElement('option');
              opt.value = loc;
              opt.textContent = loc;
              locationFilter.appendChild(opt);
            });
            // 試著還原剛剛選的場地 (如果還存在的話)
            if ([...locationFilter.options].some(o => o.value === currentSelected)) {
               locationFilter.value = currentSelected;
            }

            globalRecordsArray = Object.values(groupedData); // 把整理好的資料存進全域變數
            renderTable(); // 呼叫畫表格的函數
            
          } else {
            tableBody.innerHTML = '<tr><td colspan="7" class="p-8 text-center text-gray-400">目前尚無任何打卡紀錄</td></tr>';
            document.getElementById('stat-present').textContent = '0';
            document.getElementById('stat-left').textContent = '0';
          }
        } catch (e) {
            tableBody.innerHTML = '<tr><td colspan="7" class="p-8 text-center text-red-400">載入失敗，請稍後再試</td></tr>';
        }
      }

      // 🌟 【新增】負責把資料畫成表格與統計人數的獨立函數
      function renderTable() {
        const todayStr = new Date().toLocaleDateString('zh-TW', { timeZone: 'Asia/Taipei' });
        const selectedLocation = locationFilter.value; // 取得目前選了什麼場地
        
        // 🌟【新增】：取得搜尋框的文字並轉小寫
        const searchQuery = document.getElementById('record-search').value.toLowerCase().trim();

        let presentCount = 0;
        let leftCount = 0;

        // 🌟【升級過濾邏輯】：加入搜尋條件
        const displayRecords = globalRecordsArray.filter(g => {
          const isToday = (g.fullDate === todayStr);
          const isMatchLocation = (selectedLocation === 'ALL' || g.location === selectedLocation);
          
          // 檢查姓名、組別或副組別是否包含搜尋字串
          const isMatchSearch = !searchQuery || 
                                (g.name && g.name.toLowerCase().includes(searchQuery)) || 
                                (g.group && g.group.toLowerCase().includes(searchQuery)) ||
                                (g.subGroup && g.subGroup.toLowerCase().includes(searchQuery));

          return isToday && isMatchLocation && isMatchSearch;
        });

        

        // 統計過濾後的人數
        displayRecords.forEach(g => {
          if (g.hasIn && !g.hasOut) presentCount++;
          if (g.hasOut) leftCount++;
        });

        document.getElementById('stat-present').textContent = presentCount;
        document.getElementById('stat-left').textContent = leftCount;

        if (displayRecords.length > 0) {
          tableBody.innerHTML = displayRecords.map(g => {
return `

                <td class="p-3">${g.inTime}</td>
                
                <td class="p-3">${g.outTime}</td>
                
                <td class="p-3">
                  <div class="max-w-xs text-sm text-gray-800 font-medium">
                    ${g.bento || '<span class="text-gray-300 italic font-normal">未選擇便當</span>'}

      // 執行上傳 (或清空) 班表
      async function submitRoster(isClear = false) {
        const btn = document.getElementById(isClear ? 'clear-roster-btn' : 'upload-roster-btn');
        const originalText = btn.textContent;
        const targetDate = document.getElementById('roster-date').value;
        const rawText = document.getElementById('roster-input').value.trim();
        
        if (!targetDate) return alert('請選擇生效日期！');
        
        let rosterData = [];

        if (!isClear) {
          if (!rawText) return alert('請貼上人員名單！');
const lines = rawText.split('\n');
          for (let line of lines) {
            const cleanLine = line.trim();
            if (!cleanLine) continue; // 遇到空行直接跳過

            // 💡 智慧解析：尋找字串最尾端的 4 個數字當作密碼
            const match = cleanLine.match(/(\d{4})$/); 
            
            if (match) {
              const phone = match[1]; // 抓到的四碼
              const name = cleanLine.slice(0, -4).trim(); // 把整行扣掉最後四個字，剩下的就是名字
              
              if (name && phone) {
                rosterData.push({ name: name, phone_last4: phone });
              }
            }
          }
          if (rosterData.length === 0) return alert('解析失敗，請確認格式是否為「姓名 四碼」');
        }

        btn.textContent = '⏳ 處理中...';
        try {
          const res = await fetch(`${API_URL}/api/roster`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ password: currentPassword, targetDate, rosterData })
          });
          const data = await res.json();
          if (data.success) {
            alert(isClear ? '✅ 當日班表已清空，系統切換為自由打卡模式！' : `✅ 成功鎖定！已上傳 ${rosterData.length} 筆排班。`);
            if (!isClear) document.getElementById('roster-input').value = ''; // 成功後清空輸入框
            // 🌟【新增這行】：上傳或清空成功後，自動觸發重新抓取並刷新畫面！
            loadRosterList();
          } else {
            alert('上傳失敗：' + data.message);
          }
        } catch (e) {
          alert('伺服器連線異常');
        }
        btn.textContent = originalText;
      }

      document.getElementById('upload-roster-btn').addEventListener('click', () => submitRoster(false));
      document.getElementById('clear-roster-btn').addEventListener('click', () => submitRoster(true));


      // 🌟【防彈版 + Loading 動畫】：去後端抓資料
      async function loadRosterList() {
        const targetDate = document.getElementById('roster-date').value;
        if(!targetDate) return;
        
        // 🌟【新增】：抓到更新按鈕，並改變它的外觀
        const refreshBtn = document.getElementById('refresh-roster-btn');
        const originalText = refreshBtn.innerHTML; // 把原本的「🔄 更新名單」記下來
        refreshBtn.innerHTML = '⏳ 載入中...';
        refreshBtn.disabled = true; // 鎖住按鈕，防止狂點
        refreshBtn.classList.add('opacity-50', 'cursor-not-allowed'); // 加上半透明變灰的效果

        document.getElementById('roster-date-display').textContent = targetDate;
        const rosterUl = document.getElementById('roster-list-ul');
        const absentUl = document.getElementById('absent-list-ul');

        try {
          const res = await fetch(`${API_URL}/api/roster/list`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ password: currentPassword, targetDate })
          });
          const data = await res.json();
          
          if (data.success) {
            globalRoster = data.roster || [];
            globalAbsent = data.absent || [];
            renderRosterLists();
          } else {
            console.error('後端拒絕載入班表:', data.message);
            rosterUl.innerHTML = `<li class="text-red-400 list-none">${data.message || '拒絕訪問'}</li>`;
            absentUl.innerHTML = `<li class="text-red-400 list-none">${data.message || '拒絕訪問'}</li>`;
          }
        } catch (e) {
          console.error('🚨 班表 API 發生錯誤:', e);
          rosterUl.innerHTML = '<li class="text-red-400 list-none">載入失敗，請按 F12 查看錯誤</li>';
          absentUl.innerHTML = '<li class="text-red-400 list-none">載入失敗</li>';
        } finally {
          // 🌟【新增】：不管成功還是失敗，最後一定要執行 finally，把按鈕還原！
          refreshBtn.innerHTML = originalText;
          refreshBtn.disabled = false;
          refreshBtn.classList.remove('opacity-50', 'cursor-not-allowed');
        }
      }

      // 🌟【防彈版】：渲染與過濾班表
      function renderRosterLists() {
        try {
          const searchInput = document.getElementById('roster-search');
          const query = searchInput ? searchInput.value.toLowerCase().trim() : '';
          const rosterUl = document.getElementById('roster-list-ul');
          const absentUl = document.getElementById('absent-list-ul');

          // 🛡️ 加上 r.name && 的判斷，防止資料庫有空值導致當機
          const filteredRoster = globalRoster.filter(r => 
            (r.name && r.name.toLowerCase().includes(query)) || 
            (r.phone_last4 && r.phone_last4.includes(query))
          );
          
          const filteredAbsent = globalAbsent.filter(r => 
            (r.name && r.name.toLowerCase().includes(query)) || 
            (r.phone_last4 && r.phone_last4.includes(query))
          );

          document.getElementById('roster-count').textContent = filteredRoster.length;
          document.getElementById('absent-count').textContent = filteredAbsent.length;

          if (filteredRoster.length > 0) {
            rosterUl.innerHTML = filteredRoster.map(r => `<li>${r.name} <span class="text-gray-400 text-xs">(${r.phone_last4})</span></li>`).join('');
          } else {
            rosterUl.innerHTML = '<li class="text-gray-400 italic list-none">查無相符名單</li>';
          }
          
          if (filteredAbsent.length > 0) {
            absentUl.innerHTML = filteredAbsent.map(r => `<li class="font-bold text-red-600">${r.name} <span class="text-red-400 text-xs font-normal">(${r.phone_last4})</span></li>`).join('');
          } else if (globalRoster.length > 0 && query === '') {

      // 🌟 【新增】時效性 QR Code 生成器邏輯
      let tempQrInterval;

      // 載入場地選單給生成器使用
      async function loadLocationsForGenerator() {
        try {
          const res = await fetch(`${API_URL}/api/locations`);
          const data = await res.json();
          if (data.success) {
            document.getElementById('temp-qr-location').innerHTML = data.locations.map(loc => `<option value="${loc.id}">${loc.location_name}</option>`).join('');
          }
        } catch(e) {}
      }

      document.getElementById('generate-temp-qr-btn').addEventListener('click', async () => {
        const locId = document.getElementById('temp-qr-location').value;
        const action = document.getElementById('temp-qr-action').value;
        const duration = document.getElementById('temp-qr-duration').value;
        
        if (!locId) return alert('請先等待場地載入');

        try {
          // 🌟【升級】：改成 POST 並把密碼包在 body 裡送過去
          const res = await fetch(`${API_URL}/api/qr-token`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ 
              password: currentPassword, 
              location_id: locId, 
              duration: duration 
            })
          });
          const data = await res.json();
          
          if (data.success) {
            // 自動偵測目前網站的網址 (確保雲端也能用)
            const baseUrl = window.location.origin;
            const checkinUrl = `${baseUrl}/?loc=${data.location_id}&action=${action}&token=${data.token}`;
            
            const canvas = document.getElementById('temp-qr-canvas');
            const placeholder = document.getElementById('temp-qr-placeholder');
            const linkBox = document.getElementById('temp-qr-link-box');
            const linkInput = document.getElementById('temp-qr-link');
            const countdownText = document.getElementById('temp-qr-countdown');
            
            // 畫出條碼
            QRCode.toCanvas(canvas, checkinUrl, { width: 220, margin: 2 });
            
            // 顯示元件
            canvas.classList.remove('hidden');
            linkBox.classList.remove('hidden');
            countdownText.classList.remove('hidden');
            placeholder.classList.add('hidden');
            
            linkInput.value = checkinUrl;
            
            // 讓網址點擊可以直接複製
            linkInput.onclick = () => {
              linkInput.select();
              navigator.clipboard.writeText(linkInput.value);
              alert('網址已複製！可以貼到 LINE 群組了！');
            };
            
            // 啟動倒數計時器
            const expireTime = Date.now() + parseInt(duration);
            if (tempQrInterval) clearInterval(tempQrInterval);
            
            tempQrInterval = setInterval(() => {
              const remain = Math.floor((expireTime - Date.now()) / 1000);
              if (remain <= 0) {
                countdownText.textContent = '🛑 此條碼已過期！';
                clearInterval(tempQrInterval);
              } else {
                const h = Math.floor(remain / 3600);
                const m = Math.floor((remain % 3600) / 60);
                const s = remain % 60;
                let timeStr = h > 0 ? `${h}小時 ${m}分 ${s}秒` : `${m}分 ${s}秒`;
                countdownText.textContent = `⏳ 距離條碼失效：${timeStr}`;
              }
            }, 1000);
          }
        } catch(e) {
          alert('條碼生成失敗，請確認伺服器連線');
        }
      });
